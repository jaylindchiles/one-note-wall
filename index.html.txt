  <input id="message" placeholder="Write your one note..." />
  <button onclick="postNote()">Post</button>
</div>

<div id="board"></div>

<script>
let notes = [];

function postNote() {
  const msg = document.getElementById("message").value;
  if (!msg) return;

  notes.unshift(msg);

  document.getElementById("message").value = "";
  render();
}

function render() {
  const board = document.getElementById("board");
  board.innerHTML = "";

  notes.forEach(text => {
    const div = document.createElement("div");
    div.className = "note";
    div.innerHTML = `
      <div class="pin"></div>
      ${text}
    `;
    board.appendChild(div);
  });
}
</script>
